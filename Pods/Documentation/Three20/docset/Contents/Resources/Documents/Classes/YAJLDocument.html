Overview

JSON document interface.

@code NSData data = [NSData dataWithContentsOfFile:@“example.json”]; NSError error = nil; YAJLDocument *document = [[YAJLDocument alloc] initWithData:data parserOptions:YAJLParserOptionsNone error:&error]; // Access root element at document.root NSLog(@“Root: %@”, document.root); [document release]; @endcode

Example for streaming: @code YAJLDocument *document = [[YAJLDocument alloc] init]; document.delegate = self;

NSError *error = nil; [document parse:chunk1 error:error]; [document parse:chunk2 error:error];
